Props

You can pass props to a rule and specify the interface of props in the type parameter in ReactPeg.Rule like this:

import { ReactPeg, ActionParam } from "react-peg";

interface TextProps
{
    foo: string;
    text: string;
}

class Text extends ReactPeg.Rule<TextProps, string>{

    render()
    {
        console.log(this.props);
        return (
            <text>{`ab${this.props.text}`}</text>
        );
    }

    action(param: ActionParam)
    {
        return param.global.text();
    }
}

const parser = ReactPeg.render(<Text foo="foo" text="c"/>);
const ast = parser.parse(`abc`);
console.log(ast);

the result will be:

{ foo: 'foo', text: 'c' }
abc
